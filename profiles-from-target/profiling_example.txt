nsight profiling

nsight-systems -  (on traget command line executable)
nsys profile     --trace=cuda,osrt     --sample=cpu     --cpuctxsw=process-tree     --cuda-memory-usage=true     --duration=30     --output=~/profiles/driveseg_nsys     --force-overwrite=true     ./driveseg_object --rig=/usr/local/driveworks/data/samples/sensors/camera/camera/rig_4cam.json



get a csv report as well ::

nsys stats --report cuda_api_sum,cuda_gpu_kern_sum,cuda_gpu_mem_time_sum,osrt_sum driveseg_nsys.nsys-rep --format csv --output . --force-overwrite true
