# with chw16 format 

trtexec --onnx=/home/pomo/yolov8n-seg-lane-train2.onnx \
        --saveEngine=/home/pomo/trt-export/v8n-seg-lane-fp16_dla.trt \
        --buildDLAStandalone \
        --fp16 \
        --inputIOFormats=fp16:chw16 \
        --outputIOFormats=fp16:chw16 \
        --memPoolSize=workspace:8192
        
trtexec --loadEngine=/home/pomo/trt-export/v8n-seg-lane-fp16_dla.trt \
        --useDLACore=0 \
        --dumpLayerInfo \
        --dumpProfile \
        --exportLayerInfo=yolov8_seg_dlafp16_layers.json \
        --exportProfile=yolov8_seg_dlafp16_profile.json \
        --profilingVerbosity=detailed \
        --verbose \
        --iterations=100 \
        --avgRuns=10 \
        --warmUp=5 \
        > yolov8_seg_dlafp16_analysis.log 2>&1
